# Models

# Create a Model

Create a model

Endpoint: v1/models/new

Method: POST

Required Parameters:

Name Type Optional? Description
name string required Model Name
type string required Model Type

Example Request:

// ....

//$token = ...;
$client->post('https://ai.hellosoftware.co/api/v1/models/new', [
    'form_params' => [
        'name' => 'my model', //model name
        'type' => 'text_analysis', //no other type is supported right now
    ],
    'headers' => [
        'Authorization' => 'Bearer ' . $token,
    ],
]);

Example Response:

{
    "success":true,
    "data":{
        "name":"my model",
        "type":"text_analysis",
        "identifier":"humuQhyXDH02hyJeTHMmnw6cbCXplHqd",
        "training":false,
        "accuracy":0
    }
}

# Train a Model (SFTP)

Train a model with a dataset hosted on an SFTP server

Endpoint: v1/models/{identifier}/train

Method: POST

Required Parameters:

Name Type Optional? Description
sftp array required Array of SFTP credentials
sftp.host string required SFTP Host
sftp.port integer required SFTP Port
sftp.username string required SFTP Username
sftp.key string required SFTP SSH Key
sftp.directory string required SFTP directory containing dataset

Example Request:

// ....

//$token = ...;
//$identifier = ...;
$client->post('https://ai.hellosoftware.co/api/v1/models/' . $identifier . '/train', [
    'form_params' => [
        'sftp' => [
            'host' => '127.0.0.1',
            'port' => 22,
            'username' => 'helloai',
            // 'password' => '...'
            'key' => file_get_contents('/path/to/ssh/key'),
            'directory' => '/path/to/dataset'
        ],
    ],
    'headers' => [
        'Authorization' => 'Bearer ' . $token,
    ],
]);

Example Response:

{
    "success":true,
    "data":{
        "status":"Training 1 of 6...",
        "accuracy":0
    }
}

# Train a Model (Array)

Train a model with a dataset in an array.

Endpoint: v1/models/{identifier}/train

Method: POST

Required Parameters:

Name Type Optional? Description
dataset array required Dataset

Example Request:

// ....

//$token = ...;
//$identifier = ...;
$client->post('https://ai.hellosoftware.co/api/v1/models/' . $identifier . '/train', [
    'form_params' => [
    	'dataset' => [
    		[
    			"sample spam string", //string
    			"spam", //label
    		],
    		[
    			"is this a spam string?", //string
    			"not spam", //label
    		],
    	],
    ],
    'headers' => [
        'Authorization' => 'Bearer ' . $token,
    ],
]);

Example Response:

{
    "success":true,
    "data":{
        "status":"Training 1 of 6...",
        "accuracy":0
    }
}

# Predict against a Model

Predict against a model

Endpoint: v1/models/{identifier}/predict

Method: POST

Required Parameters:

Name Type Optional? Description
dataset array required Array of items

Example Request:

// ....

//$token = ...;
//$identifier = ...;
$client->post('https://ai.hellosoftware.co/api/v1/models/' . $identifier . '/train', [
	'form_params' => [
		'dataset' => ['Is this a spam string?'], // pass in an array because helloai allows multiple predictions at once
	],
    'headers' => [
        'Authorization' => 'Bearer ' . $token,
    ],
]);

Example Response:

{
    "success":true,
    "data":{
        "status":["not spam"],
        "input":["is this a spam string?"]
    }
}
